
#Super pour voir où on se trouve, où en est le HEAD, en couleur
git config --global alias.lol "log --oneline --graph --decorate --all"

#Pour voir differents renseignement sur le depot distant
git config --global alias.rem "remote show origin"

#Se placer sur une  branche en un clin d'oeil
git config --global alias.co checkout

#Nouvelle branche [nom] ou liste des braches utilisé seul
git config --global alias.br branch

#Commiter
git config --global alias.ci commit

#ou on en est...
git config --global alias.st status
